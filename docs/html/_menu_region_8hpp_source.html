281 void MenuRegion<Item>::set_selectables(gch::vector<short> selects) {
282 // Update our rows
283 rows = std::move(selects);
284
285 if (selector != nullptr) {
286 // If we have no selectable rows, deactivate and return
287 if (rows.empty()) {
288 selector->deactivate();
289 return;
290 }
291 // Try to find the first row that actually has selectable items
292 bool foundValid = false;
293 for (int i = 0; i < rows.size(); ++i) {
294 int actualRow = rows[i];
295 // Check bounds and row content
296 if (actualRow >= 0 &&
297 actualRow < (int)menus.size() &&
298 menus[actualRow] &&
299 !menus[actualRow]->empty()) {
300 s.row = i; // found a valid row for the selector
301 foundValid = true;
302 break;
303 }
304 }
305 // If none of the selectable regions have any valid options,
306 // deactivate the selector.
307 if (!foundValid) {
308 selector->deactivate();
309 return;
310 }
